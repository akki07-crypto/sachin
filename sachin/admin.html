// Guard Page Access const token = localStorage.getItem('adminToken'); if (!token) { window.location.href = 'login.html'; } async function fetchQueries() { const tableBody = document.getElementById('queries-body'); const totalCount = document.getElementById('total-queries-count'); const todayCount = document.getElementById('today-queries-count'); try { const response = await fetch('/api/contacts', { headers: { 'Authorization': `Bearer ${token}` } }); if (response.status === 401) { localStorage.removeItem('adminToken'); window.location.href = 'login.html'; return; } if (!response.ok) throw new Error('Failed to fetch data'); const queries = await response.json(); // Update total stat totalCount.innerText = queries.length; // Count today's stats const today = new Date().toDateString(); const todayList = queries.filter(q => new Date(q.date).toDateString() === today); todayCount.innerText = todayList.length; // Sort descending by date (newest first) queries.sort((a, b) => new Date(b.date) - new Date(a.date)); if (queries.length === 0) { tableBody.innerHTML = `

No inquiries found in database yet.

`; return; } tableBody.innerHTML = ''; queries.forEach(q => { const row = document.createElement('tr'); const localDate = new Date(q.date).toLocaleString(); row.innerHTML = ` ${escapeHTML(q.name)} ${escapeHTML(q.email)} ${escapeHTML(q.phone)}
${escapeHTML(q.message)}
${localDate} `; tableBody.appendChild(row); }); } catch (err) { console.error(err); tableBody.innerHTML = `

Error loading inquiries from server.

`; } } function logout() { localStorage.removeItem('adminToken'); localStorage.removeItem('adminUser'); window.location.href = 'login.html'; } function escapeHTML(str) { if (!str) return ''; return str.replace(/[&<>'"]/g, tag => ({ '&': '&', '<': '<', '>': '>', "'": ''', '"': '"' }[tag] || tag) ); } // Fetch on load if (token) { fetchQueries(); }
